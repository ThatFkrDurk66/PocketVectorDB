USAGE EXAMPLE:
--------------

  from pocketvectordb import VectorDB
  import numpy as np
  
  # Create database
  db = VectorDB("~/my_vectors", dimension=128)
  
  # Add vectors
  vec = np.random.rand(128)
  db.add(vec, metadata={"source": "test"}, text="Hello")
  
  # Search
  results = db.query(vec, n_results=5)
  print(results)
  
  # Save
  db.save()
